cmake_minimum_required(VERSION 3.24)
project(Dungeon LANGUAGES C)

# Builds ONLY the platform exe (the stable binary). The reloadable game dll
# is built by reload.bat with a single gcc call -- that is the iteration loop.
# MinGW gcc required: seni's layout embedding uses GNU .incbin assembly.

set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)

set(VENDOR_DIR "${CMAKE_CURRENT_SOURCE_DIR}/../vendor")
set(ENGINE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/../engine")
set(SENI_DIR "C:/Users/andres/Development/seni" CACHE PATH "Path to the seni library checkout")
set(KANSI_DIR "C:/Users/andres/Development/kansi" CACHE PATH "Path to the kansi library checkout")

list(APPEND CMAKE_PREFIX_PATH "${VENDOR_DIR}/SDL3-mingw/x86_64-w64-mingw32")
find_package(SDL3 CONFIG REQUIRED)

add_executable(dungeon
  "${ENGINE_DIR}/src/platform/win32_main.c"
  "${SENI_DIR}/seni.c"
  "${SENI_DIR}/arena.c"
  "${KANSI_DIR}/kansi.c"
  "${KANSI_DIR}/platform_windows.c")

target_include_directories(dungeon PRIVATE src "${ENGINE_DIR}/src" "${SENI_DIR}" "${KANSI_DIR}")
target_link_libraries(dungeon PRIVATE SDL3::SDL3)
target_compile_options(dungeon PRIVATE -Wall -Wextra)

add_custom_command(TARGET dungeon POST_BUILD
  COMMAND "${CMAKE_COMMAND}" -E copy_if_different
    "${VENDOR_DIR}/SDL3-mingw/x86_64-w64-mingw32/bin/SDL3.dll"
    "$<TARGET_FILE_DIR:dungeon>/SDL3.dll")
